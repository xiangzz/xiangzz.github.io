class PCQueue<T> {
    private final LinkedList<T> q = new LinkedList<>();
    private final int cap;
    private final ReentrantLock lock = new ReentrantLock();

    public void put(T x) throws InterruptedException {

            q.add(x);
            notEmpty.signal();
        } finally { lock.unlock(); }

            return x;
        } finally { lock.unlock(); }
    }
}
