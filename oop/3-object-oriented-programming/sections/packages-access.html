6. 包和访问控制

6.1 包的概念

包（Package）是Java中组织类和接口的机制，类似于文件系统中的文件夹。包提供了命名空间，避免类名冲突，并且便于管理大型项目。

// 包声明必须是文件的第一行（除了注释）
package com.example.util;

import java.util.Date;
import java.util.List;
import java.util.ArrayList;

public class DateUtil {
    public static String getCurrentDate() {
        Date now = new Date();
        return now.toString();
    }
    
    public static List getWeekDays() {
        List days = new ArrayList<>();
        days.add("Monday");
        days.add("Tuesday");
        days.add("Wednesday");
        days.add("Thursday");
        days.add("Friday");
        return days;
    }
}

Java包的层次结构

6.2 包的声明和导入

使用package关键字声明包，使用import关键字导入其他包中的类。包名通常采用域名倒置的命名规则。

// 文件：com/example/model/Student.java
package com.example.model;

public class Student {
    private String name;
    private int age;
    
    public Student(String name, int age) {
        this.name = name;
        this.age = age;
    }
    
    // getter和setter方法
    public String getName() { return name; }
    public void setName(String name) { this.name = name; }
    public int getAge() { return age; }
    public void setAge(int age) { this.age = age; }
}

// 文件：com/example/service/StudentService.java
package com.example.service;

import com.example.model.Student;  // 导入Student类
import java.util.*;  // 导入java.util包下的所有类

public class StudentService {
    private List students;
    
    public StudentService() {
        this.students = new ArrayList<>();
    }
    
    public void addStudent(Student student) {
        students.add(student);
    }
    
    public List getAllStudents() {
        return new ArrayList<>(students);
    }
}

6.3 访问权限修饰符

Java提供了四种访问权限修饰符，用于控制类、方法和变量的访问范围。

package com.example.access;

public class AccessExample {
    public String publicField = "公共字段";        // 任何地方都可以访问
    protected String protectedField = "受保护字段"; // 同包或子类可以访问
    String packageField = "包私有字段";            // 同包内可以访问（默认访问权限）
    private String privateField = "私有字段";      // 只有本类内可以访问
    
    // 公共方法
    public void publicMethod() {
        System.out.println("这是公共方法");
    }
    
    // 受保护方法
    protected void protectedMethod() {
        System.out.println("这是受保护方法");
    }
    
    // 包私有方法（默认访问权限）
    void packageMethod() {
        System.out.println("这是包私有方法");
    }
    
    // 私有方法
    private void privateMethod() {
        System.out.println("这是私有方法");
    }
    
    // 在同一个类中，所有成员都可以访问
    public void testAccess() {
        System.out.println(publicField);
        System.out.println(protectedField);
        System.out.println(packageField);
        System.out.println(privateField);  // 私有字段也可以访问
        
        publicMethod();
        protectedMethod();
        packageMethod();
        privateMethod();  // 私有方法也可以调用
    }
}

6.4 访问权限测试

通过实际例子来理解不同访问权限的作用范围。

// 同包内的类
package com.example.access;

public class SamePackageTest {
    public void testAccess() {
        AccessExample example = new AccessExample();
        
        // 可以访问public、protected、package字段
        System.out.println(example.publicField);     // ✓ 可以访问
        System.out.println(example.protectedField);  // ✓ 可以访问
        System.out.println(example.packageField);    // ✓ 可以访问
        // System.out.println(example.privateField); // ✗ 编译错误
        
        // 可以调用public、protected、package方法
        example.publicMethod();     // ✓ 可以调用
        example.protectedMethod();  // ✓ 可以调用
        example.packageMethod();    // ✓ 可以调用
        // example.privateMethod();  // ✗ 编译错误
    }
}

// 不同包内的类
package com.example.other;

import com.example.access.AccessExample;

public class DifferentPackageTest {
    public void testAccess() {
        AccessExample example = new AccessExample();
        
        // 只能访问public字段
        System.out.println(example.publicField);      // ✓ 可以访问
        // System.out.println(example.protectedField); // ✗ 编译错误
        // System.out.println(example.packageField);   // ✗ 编译错误
        // System.out.println(example.privateField);   // ✗ 编译错误
        
        // 只能调用public方法
        example.publicMethod();      // ✓ 可以调用
        // example.protectedMethod(); // ✗ 编译错误
        // example.packageMethod();   // ✗ 编译错误
        // example.privateMethod();   // ✗ 编译错误
    }
}

// 不同包内的子类
package com.example.other;

import com.example.access.AccessExample;

public class SubclassTest extends AccessExample {
    public void testAccess() {
        // 子类可以访问public和protected成员
        System.out.println(publicField);     // ✓ 可以访问
        System.out.println(protectedField);  // ✓ 可以访问
        // System.out.println(packageField);  // ✗ 编译错误
        // System.out.println(privateField);  // ✗ 编译错误
        
        publicMethod();     // ✓ 可以调用
        protectedMethod();  // ✓ 可以调用
        // packageMethod();  // ✗ 编译错误
        // privateMethod();  // ✗ 编译错误
    }
}

6.5 封装的最佳实践

良好的封装设计应该遵循最小权限原则，只暴露必要的接口，隐藏内部实现细节。

package com.example.encapsulation;

public class BankAccount {
    // 私有字段，外部无法直接访问
    private String accountNumber;
    private String accountHolder;
    private double balance;
    private static final double MIN_BALANCE = 0.0;
    
    // 公共构造方法
    public BankAccount(String accountNumber, String accountHolder, double initialBalance) {
        this.accountNumber = accountNumber;
        this.accountHolder = accountHolder;
        this.balance = Math.max(initialBalance, MIN_BALANCE);
    }
    
    // 公共getter方法，提供只读访问
    public String getAccountNumber() {
        return accountNumber;
    }
    
    public String getAccountHolder() {
        return accountHolder;
    }
    
    public double getBalance() {
        return balance;
    }
    
    // 公共业务方法，控制数据修改
    public boolean deposit(double amount) {
        if (amount > 0) {
            balance += amount;
            return true;
        }
        return false;
    }
    
    public boolean withdraw(double amount) {
        if (amount > 0 && balance >= amount) {
            balance -= amount;
            return true;
        }
        return false;
    }
    
    // 私有辅助方法，内部使用
    private void logTransaction(String type, double amount) {
        System.out.println("账户 " + accountNumber + " " + type + " " + amount);
    }
    
    // 受保护方法，供子类使用
    protected boolean isValidAmount(double amount) {
        return amount > 0 && amount <= 10000;  // 单次操作限额
    }
}

// 使用示例
class BankAccountTest {
    public static void main(String[] args) {
        BankAccount account = new BankAccount("123456789", "张三", 1000.0);
        
        // 通过公共方法访问和修改数据
        System.out.println("账户余额：" + account.getBalance());
        
        account.deposit(500.0);
        System.out.println("存款后余额：" + account.getBalance());
        
        account.withdraw(200.0);
        System.out.println("取款后余额：" + account.getBalance());
        
        // 无法直接访问私有字段
        // account.balance = 999999;  // 编译错误
    }
}
