import java.util.Arrays;

    // 指定范围的随机数
    public static void demonstrateRangeRandom() {
        System.out.println("\n=== 指定范围的随机数 ===");
        Random random = new Random();
        
        // 生成指定范围的整数
        int min = 10, max = 50;
        for (int i = 0; i < 5; i++) {
            int randomInRange = random.nextInt(max - min + 1) + min;
            System.out.println(min + "-" + max + "之间的随机数: " + randomInRange);
        }
        
        // 生成指定范围的浮点数
        double minDouble = 1.5, maxDouble = 9.8;
        for (int i = 0; i < 3; i++) {
            double randomDouble = minDouble + (maxDouble - minDouble) * random.nextDouble();
            System.out.printf("%.1f-%.1f之间的随机数: %.2f%n", minDouble, maxDouble, randomDouble);
        }
    }
    
    // 随机数组和集合
    public static void demonstrateRandomArrays() {
        System.out.println("\n=== 随机数组生成 ===");
        Random random = new Random();
        
        // 生成随机整数数组

    // 高斯分布随机数
    public static void demonstrateGaussianRandom() {
        System.out.println("\n=== 高斯分布随机数 ===");
        Random random = new Random();
        
        System.out.println("标准正态分布随机数（均值0，标准差1）:");
        for (int i = 0; i < 5; i++) {
            double gaussian = random.nextGaussian();
            System.out.printf("%.3f ", gaussian);
        }
        System.out.println();
        
        // 自定义均值和标准差的正态分布
        double mean = 100;  // 均值
        double stdDev = 15; // 标准差
        System.out.println("\n自定义正态分布（均值100，标准差15）:");
        for (int i = 0; i < 5; i++) {
            double customGaussian = mean + stdDev * random.nextGaussian();
            System.out.printf("%.1f ", customGaussian);

            System.out.print(dice + " ");
        }
        System.out.println();

        String[] options = {"苹果", "香蕉", "橙子", "葡萄", "草莓"};

    // 生成不重复的随机数
    public static int[] generateUniqueRandomNumbers(int min, int max, int count) {
        if (count > (max - min + 1)) {
            throw new IllegalArgumentException("无法生成足够的不重复随机数");
        }
        
        Random random = new Random();
        int[] result = new int[count];
        boolean[] used = new boolean[max - min + 1];
        
        for (int i = 0; i < count; i++) {
            int num;
            do {
                num = random.nextInt(max - min + 1) + min;
            } while (used[num - min]);
            
            used[num - min] = true;
            result[i] = num;
        }
        
        return result;
    }
    
    // 随机打乱数组
    public static void shuffleArray(Object[] array) {
        Random random = new Random();
        for (int i = array.length - 1; i > 0; i--) {
            int j = random.nextInt(i + 1);
            Object temp = array[i];
            array[i] = array[j];
            array[j] = temp;
        }
    }
    
    // 生成随机密码
    public static String generateRandomPassword(int length) {
        String chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789!@#$%^&*";
        Random random = new Random();
        StringBuilder password = new StringBuilder();
        
        for (int i = 0; i < length; i++) {
            password.append(chars.charAt(random.nextInt(chars.length())));
        }
        
        return password.toString();
    }
    
    // 生成随机颜色
    public static String generateRandomColor() {
        Random random = new Random();
        int r = random.nextInt(256);
        int g = random.nextInt(256);
        int b = random.nextInt(256);
        return String.format("RGB(%d, %d, %d)", r, g, b);
    }

        System.out.println("当前工作目录: " + System.getProperty("user.dir"));
        
        // 环境变量
        String path = System.getenv("PATH");
        if (path != null) {
            System.out.println("PATH环境变量长度: " + path.length());
        }
        
        // 时间测量
        long startTime = System.currentTimeMillis();
        // 模拟一些操作
        try {
            Thread.sleep(100);
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
        long endTime = System.currentTimeMillis();
        System.out.println("操作耗时: " + (endTime - startTime) + "毫秒");
        
        // 纳秒时间（更精确）
        long nanoStart = System.nanoTime();
        int sum = 0;
        for (int i = 0; i < 1000000; i++) {
            sum += i;
        }
        long nanoEnd = System.nanoTime();
        System.out.println("循环计算耗时: " + (nanoEnd - nanoStart) + "纳秒");
        
        // 数组复制
        int[] source = {1, 2, 3, 4, 5};
        int[] dest = new int[5];
        System.arraycopy(source, 0, dest, 0, source.length);
        System.out.println("数组复制结果: " + Arrays.toString(dest));
    }
    
    // 自定义工具类示例
    public static void demonstrateCustomUtils() {
        System.out.println("\n=== 自定义工具类示例 ===");
        
        // 字符串工具
        System.out.println("是否为空: " + StringUtils.isEmpty(""));
        System.out.println("是否为空: " + StringUtils.isEmpty("Hello"));
        System.out.println("首字母大写: " + StringUtils.capitalize("hello world"));
        
        // 数组工具
        int[] array = {1, 2, 3, 4, 5};
        System.out.println("数组包含3: " + ArrayUtils.contains(array, 3));
        System.out.println("数组包含6: " + ArrayUtils.contains(array, 6));
        
        // 数学工具
        System.out.println("是否为偶数: " + MathUtils.isEven(4));
        System.out.println("是否为质数: " + MathUtils.isPrime(17));
        System.out.println("阶乘5!: " + MathUtils.factorial(5));
    }
    
    // 自定义字符串工具类
    static class StringUtils {
        public static boolean isEmpty(String str) {
            return str == null || str.length() == 0;
        }
        
        public static String capitalize(String str) {
            if (isEmpty(str)) return str;
            return str.substring(0, 1).toUpperCase() + str.substring(1);
        }
    }
    
    // 自定义数组工具类
    static class ArrayUtils {
        public static boolean contains(int[] array, int value) {
            for (int item : array) {
                if (item == value) return true;
            }
            return false;
        }
    }
    
    // 自定义数学工具类
    static class MathUtils {
        public static boolean isEven(int number) {
            return number % 2 == 0;
        }
        
        public static boolean isPrime(int number) {
            if (number < 2) return false;
            for (int i = 2; i <= Math.sqrt(number); i++) {
                if (number % i == 0) return false;
            }
            return true;
        }
        
        public static long factorial(int n) {
            if (n < 0) throw new IllegalArgumentException("负数没有阶乘");
            long result = 1;
            for (int i = 2; i <= n; i++) {
                result *= i;
            }
            return result;
        }
    }
