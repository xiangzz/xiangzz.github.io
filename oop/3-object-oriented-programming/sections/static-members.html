5. 静态变量和静态方法

5.1 静态变量

静态变量（类变量）属于类本身，而不是某个特定的对象。所有对象共享同一个静态变量，使用static关键字声明。

public class Counter {
    private static int count = 0;  // 静态变量
    private int instanceId;        // 实例变量
    
    public Counter() {
        count++;  // 每创建一个对象，静态变量count增加1
        this.instanceId = count;
    }
    
    public static int getCount() {
        return count;
    }
    
    public int getInstanceId() {
        return instanceId;
    }
    
    public static void main(String[] args) {
        System.out.println("初始计数：" + Counter.getCount());
        
        Counter c1 = new Counter();
        Counter c2 = new Counter();
        Counter c3 = new Counter();
        
        System.out.println("创建3个对象后的计数：" + Counter.getCount());
        System.out.println("c1的ID：" + c1.getInstanceId());
        System.out.println("c2的ID：" + c2.getInstanceId());
        System.out.println("c3的ID：" + c3.getInstanceId());
    }
}

5.2 静态方法

静态方法属于类本身，可以直接通过类名调用，不需要创建对象。静态方法中不能直接访问非静态成员。

public class MathHelper {
    public static final double PI = 3.14159;  // 静态常量
    
    // 静态方法：计算圆的面积
    public static double calculateCircleArea(double radius) {
        return PI * radius * radius;
    }
    
    // 静态方法：计算矩形面积
    public static double calculateRectangleArea(double width, double height) {
        return width * height;
    }
    
    // 静态方法：判断是否为偶数
    public static boolean isEven(int number) {
        return number % 2 == 0;
    }
    
    // 静态方法：获取两个数的最大值
    public static int max(int a, int b) {
        return a > b ? a : b;
    }
    
    public static void main(String[] args) {
        // 直接通过类名调用静态方法
        System.out.println("圆面积：" + MathHelper.calculateCircleArea(5));
        System.out.println("矩形面积：" + MathHelper.calculateRectangleArea(4, 6));
        System.out.println("10是偶数吗？" + MathHelper.isEven(10));
        System.out.println("最大值：" + MathHelper.max(15, 20));
    }
}

5.3 静态代码块

静态代码块在类第一次被加载时执行，且只执行一次。通常用于初始化静态变量或执行一次性的初始化操作。

public class DatabaseConfig {
    private static String databaseUrl;
    private static String username;
    private static String password;
    
    // 静态代码块
    static {
        System.out.println("正在初始化数据库配置...");
        databaseUrl = "jdbc:mysql://localhost:3306/mydb";
        username = "admin";
        password = "password123";
        System.out.println("数据库配置初始化完成");
    }
    
    // 实例代码块
    {
        System.out.println("创建DatabaseConfig实例");
    }
    
    public DatabaseConfig() {
        System.out.println("DatabaseConfig构造方法执行");
    }
    
    public static void printConfig() {
        System.out.println("数据库URL: " + databaseUrl);
        System.out.println("用户名: " + username);
    }
    
    public static void main(String[] args) {
        System.out.println("程序开始执行");
        
        // 第一次访问类时，静态代码块执行
        DatabaseConfig.printConfig();
        
        System.out.println("\n创建第一个对象：");
        DatabaseConfig config1 = new DatabaseConfig();
        
        System.out.println("\n创建第二个对象：");
        DatabaseConfig config2 = new DatabaseConfig();
    }
}

5.4 静态导入

静态导入允许直接使用静态成员，而不需要类名前缀。使用import static语句实现。

import static java.lang.Math.*;  // 静态导入Math类的所有静态成员
import static java.lang.System.out;  // 静态导入System.out

public class StaticImportExample {
    public static void main(String[] args) {
        // 不需要Math.前缀
        double result1 = sqrt(16);  // 相当于Math.sqrt(16)
        double result2 = pow(2, 3);  // 相当于Math.pow(2, 3)
        double result3 = max(10, 20);  // 相当于Math.max(10, 20)
        
        // 不需要System.out前缀
        out.println("平方根：" + result1);  // 相当于System.out.println
        out.println("幂运算：" + result2);
        out.println("最大值：" + result3);
        
        // 使用Math常量
        out.println("圆周率：" + PI);  // 相当于Math.PI
        out.println("自然对数底：" + E);  // 相当于Math.E
    }
}

5.5 单例模式

单例模式是一种设计模式，确保一个类只有一个实例，并提供全局访问点。常使用静态变量和静态方法实现。

// 饿汉式单例模式
class EagerSingleton {
    // 静态变量，类加载时就创建实例
    private static final EagerSingleton instance = new EagerSingleton();
    
    // 私有构造方法，防止外部创建实例
    private EagerSingleton() {
        System.out.println("EagerSingleton实例创建");
    }
    
    // 静态方法，提供全局访问点
    public static EagerSingleton getInstance() {
        return instance;
    }
    
    public void doSomething() {
        System.out.println("EagerSingleton执行操作");
    }
}

// 懒汉式单例模式
class LazySingleton {
    private static LazySingleton instance;
    
    private LazySingleton() {
        System.out.println("LazySingleton实例创建");
    }
    
    // 线程安全的懒汉式
    public static synchronized LazySingleton getInstance() {
        if (instance == null) {
            instance = new LazySingleton();
        }
        return instance;
    }
    
    public void doSomething() {
        System.out.println("LazySingleton执行操作");
    }
}

public class SingletonTest {
    public static void main(String[] args) {
        // 测试饿汉式单例
        EagerSingleton eager1 = EagerSingleton.getInstance();
        EagerSingleton eager2 = EagerSingleton.getInstance();
        System.out.println("饿汉式是否为同一实例：" + (eager1 == eager2));
        
        // 测试懒汉式单例
        LazySingleton lazy1 = LazySingleton.getInstance();
        LazySingleton lazy2 = LazySingleton.getInstance();
        System.out.println("懒汉式是否为同一实例：" + (lazy1 == lazy2));
    }
}
